def es_par(numero):
    return numero % 2 == 0

def determinar_signo(numero):
    if numero > 0:
        return "positivo"
    elif numero < 0:
        return "negativo"
    else:
        return "cero"



def calcular_cuadrado(numero):
    return numero * numero

    print(numero1, "es par:", es_par(numero1))

    numero3 = 5
