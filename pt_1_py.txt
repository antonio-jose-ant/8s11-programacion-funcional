def identificar_tipo_dato(dato):
    if dato == "True" or dato == "False":
        return "El dato es de tipo booleano."
    elif dato.isdigit():
        return "El dato es de tipo entero."
    else:
        return "El dato es de tipo cadena de texto."


dato = input("Ingrese un dato: ")

print(identificar_tipo_dato(dato))
