import math

def es_par(numero):
    return numero % 2 == 0

def calcular_area_circulo(radio):
    return math.pi * radio ** 2

def convertir_a_mayusculas(cadena):
    return cadena.upper()

def main():
    numero = 7
    print(f"{numero} es par: {es_par(numero)}")

    radio = 5.0
    print(f"El área del círculo con radio {radio} es: {calcular_area_circulo(radio)}")

    cadena_minusculas = "Hola Mundo"
    cadena_mayusculas = convertir_a_mayusculas(cadena_minusculas)
    print(f"Cadena en mayúsculas: {cadena_mayusculas}")

if __name__ == "__main__":
    main()
