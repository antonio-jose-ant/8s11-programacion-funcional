package funciones;

/**
 *
 * @author judit
 */
public class Funciones {

    // Función pura para determinar si un número es par o impar
    public static boolean esPar(int numero) {
        return numero % 2 == 0;
    }

    // Función pura para calcular el área de un círculo
    public static double calcularAreaCirculo(double radio) {
        return Math.PI * Math.pow(radio, 2);
    }

    // Función pura para convertir una cadena a mayúsculas
    public static String convertirAMayusculas(String cadena) {
        return cadena.toUpperCase();
    }

    // Función principal que utiliza las funciones puras anteriores
    public static void main(String[] args) {
        int numero = 7;
        System.out.println(numero + " es par: " + esPar(numero));

        double radio = 5.0;
        System.out.println("El área del círculo con radio " + radio + " es: " + calcularAreaCirculo(radio));

        String cadenaMinusculas = "Hola Mundo";
        String cadenaMayusculas = convertirAMayusculas(cadenaMinusculas);
        System.out.println("Cadena en mayúsculas: " + cadenaMayusculas);
    }
}
