
# Definición de la función suma que toma dos números enteros y devuelve la suma
def suma(a, b):
    return a + b

# Definición de la función resta que toma dos números enteros y devuelve la resta
def resta(a, b):
    return a - b

# Definición de la función división que toma dos números enteros y devuelve el cociente
# Se incluye un control para evitar la división por cero
def division(a, b):
    if b != 0:
        return a / b
    else:
        print("No se puede dividir por cero.")
        return 0

# Definición de la función combinacion_de_funciones que llama a las funciones anteriores y muestra los resultados
def combinacion_de_funciones():
    num1 = 10
    num2 = 5

    # Llamamos a las funciones y guardamos los resultados
    resultado_suma = suma(num1, num2)
    resultado_resta = resta(num1, num2)
    resultado_division = division(num1, num2)

    # Imprimimos los resultados
    print("El resultado de la suma es:", resultado_suma)
    print("El resultado de la resta es:", resultado_resta)
    print("El resultado de la división es:", resultado_division)

# Bloque principal que verifica si este script es el programa principal y llama a la función combinacion_de_funciones
if __name__ == "__main__":
    combinacion_de_funciones()
