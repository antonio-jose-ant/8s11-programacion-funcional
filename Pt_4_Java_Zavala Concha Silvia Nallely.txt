
public class p2 {
    public static void main(String[] args) {
        // Llamamos a la función combinada
        combinacionDeFunciones();
    }

    // Función que suma dos números enteros
    public static int suma(int a, int b) {
        return a + b;
    }

    // Función que resta dos números enteros
    public static int resta(int a, int b) {
        return a - b;
    }

    // Función que divide dos números enteros
    public static int division(int a, int b) {
        // Asumimos que b no es cero para evitar una división por cero
        if (b != 0) {
            return a / b;
        } else {
            // Manejar la división por cero de manera apropiada
            System.out.println("No se puede dividir por cero.");
            return 0;
        }
    }

    // Función que llama a las tres funciones anteriores y hace algo con los resultados
    public static void combinacionDeFunciones() {
        int num1 = 10;
        int num2 = 5;

        // Llamamos a las tres funciones y guardamos los resultados
        int resultadoSuma = suma(num1, num2);
        int resultadoResta = resta(num1, num2);
        int resultadoDivision = division(num1, num2);

        // Imprimimos los resultados
        System.out.println("El resultado de la suma es: " + resultadoSuma);
        System.out.println("El resultado de la resta es: " + resultadoResta);
        System.out.println("El resultado de la división es: " + resultadoDivision);

        
    }
}
