def mostrar_lista(lista):
    print("Lista actual:", lista)

# Lista predefinida
lista_numeros = [7, 2, 11, 5, 9]
mostrar_lista(lista_numeros)

elemento = int(input("Ingrese el elemento que desea agregar: "))
lista_numeros.append(elemento)
mostrar_lista(lista_numeros)

elemento = int(input("Ingrese el elemento que desea eliminar: "))
if elemento in lista_numeros:
    lista_numeros.remove(elemento)
    print(f"Elemento {elemento} eliminado de la lista.")
    mostrar_lista(lista_numeros)
else:
    print("El elemento no está en la lista.")

lista_numeros.sort()
print("Lista ordenada en orden ascendente:")
mostrar_lista(lista_numeros)

elemento = int(input("Ingrese el elemento que desea buscar: "))
if elemento in lista_numeros:
    indice = lista_numeros.index(elemento)
    print(f"El elemento {elemento} se encuentra en el índice {indice} de la lista.")
else:
    print("El elemento no está en la lista.")
