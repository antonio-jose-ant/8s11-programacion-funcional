//arboles

class Nodo:
    def __init__(self, valor):
        self.valor = valor
        self.izquierda = None
        self.derecha = None

def inser_nod(raiz, valor):
    if raiz is None:
        return Nodo(valor)
    else:
        if valor < raiz.valor:
            raiz.izquierda = inser_nod(raiz.izquierda, valor)
        else:
            raiz.derecha = inser_nod(raiz.derecha, valor)
    return raiz

def preorden(raiz):
    if raiz:
        print(raiz.valor, end=" ")
        preorden(raiz.izquierda)
        preorden(raiz.derecha)

def inorden(raiz):
    if raiz:
        inorden(raiz.izquierda)
        print(raiz.valor, end=" ")
        inorden(raiz.derecha)

def postorden(raiz):
    if raiz:
        postorden(raiz.izquierda)
        postorden(raiz.derecha)
        print(raiz.valor, end=" ")

def const_arb():
    cantidad_nodos = int(input("Ingrese el número de nodos: "))
    raiz = None
    for _ in range(cantidad_nodos):
        valor = int(input("Ingrese el valor: "))
        raiz = inser_nod(raiz, valor)
    return raiz

arbol = const_arb()
print("\nEl árbol creado es:")
print("\nRecorrido en Preorden:")
preorden(arbol)
print("\nRecorrido en Inorden:")
inorden(arbol)
print("\nRecorrido en Postorden:")
postorden(arbol)
