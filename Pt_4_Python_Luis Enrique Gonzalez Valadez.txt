listas = [100, 214, 656, 897, 342]
listas.append(4342)
listas.remove(4342)


listas.sort()
print(listas)

def busqueda(elemento, lista):
    return elemento in lista
resultado = 100
encontrado = busqueda(resultado, listas)
if encontrado:
    print(f"'{resultado}' encontrado.")
else:
    print(f"'{resultado}' no encontrado.")
