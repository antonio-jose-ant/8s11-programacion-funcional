def funcion_pura1(x):
    return x + 2

def funcion_pura2(y):
    return y * 3

def funcion_pura3(z):
    return z + z

def funcion_combinada(input1, input2):
    resultado1 = funcion_pura1(input1)
    resultado2 = funcion_pura2(input2)
    resultado3 = funcion_pura3(resultado2 + resultado1)
    return resultado3

if __name__ == "__main__":
    valor1 = int(input("Ingrese el primer valor: "))
    valor2 = int(input("Ingrese el segundo valor: "))
    resultado_final = funcion_combinada(valor1, valor2)
    print("El resultado final es:", resultado_final)
