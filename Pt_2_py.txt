//Intervalos

def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

array = [10, 20, 30, 40, 50]

# Solicitar al usuario ingresar el número a buscar
target = int(input("Ingrese el número que desea buscar: "))

result = linear_search(array, target)
if result != -1:
    print(f"El elemento {target} fue encontrado en el índice {result}.")
else:
    print(f"El elemento {target} no fue encontrado en el array.")
