package data;

/**
 *
 * @author jorge
 */
import java.util.Scanner;

public class Data {
    
    // Función para verificar si una cadena representa un número entero
    public static boolean esNumeroEntero(String cadena) {
        try {
            Integer.parseInt(cadena);
            return true;
        } catch (NumberFormatException e) {
            return false;
        }
    }
    
    // Función para verificar si una cadena representa un número flotante
    public static boolean esNumeroFlotante(String cadena) {
        try {
            Double.parseDouble(cadena);
            return true;
        } catch (NumberFormatException e) {
            return false;
        }
    }
    
    // Función para verificar si una cadena representa un booleano
    public static boolean esBooleano(String cadena) {
        return cadena.equalsIgnoreCase("true") || cadena.equalsIgnoreCase("false");
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.println("Ingrese un dato:");
        String entrada = scanner.nextLine();

        String tipo;
        
        // Verificar el tipo de dato
        if (esNumeroEntero(entrada)) {
            tipo = "número entero";
        } else if (esNumeroFlotante(entrada)) {
            tipo = "número flotante";
        } else if (esBooleano(entrada)) {
            tipo = "booleano";
        } else {
            tipo = "cadena";
        }

        System.out.println("El dato ingresado es de tipo: " + tipo);

        scanner.close();
    }
}
