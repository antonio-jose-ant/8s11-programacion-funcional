def suma(a, b):
    return a + b

def resta(a, b):
    return a - b

def multiplicacion(a, b):
    return a * b

def realizarOperaciones():
    num1 = float(input("Ingresa un número: "))
    num2 = float(input("Ingresa otro número: "))

    sumar = suma(num1, num2)
    restar = resta(num1, num2)
    multiplicar = multiplicacion(num1, num2)

    print(f"El resultado de la suma es: {sumar}")
    print(f"El resultado de la resta es: {restar}")
    print(f"El resultado de la multiplicación es: {multiplicar}")

realizarOperaciones()
