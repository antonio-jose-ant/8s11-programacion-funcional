def main():
    entrada = input("Ingresa un dato: ")

    if entrada.isdigit():
        print("Tipo entero (int).")
    elif entrada.replace(".", "", 1).isdigit():
        print("Tipo decimal (float).")
    elif entrada.lower() == "true" or entrada.lower() == "false":
        print("Tipo booleano (bool).")
    else:
        print("Tipo cadena (str).")

if __name__ == "__main__":
    main()
