def sumar(a, b):
    return a + b

def multiplicar(a, b):
    return a * b

def es_par(numero):
    return numero % 2 == 0

def es_producto_par(a, b):
    resultado = multiplicar(a, b)
    return es_par(resultado)

def main():
    num1 = 5
    num2 = 3

    suma = sumar(num1, num2)
    producto = multiplicar(num1, num2)
    es_par_producto = es_producto_par(num1, num2)

    print("La suma es:", suma)
    print("El producto es:", producto)
    print("Â¿El producto es par?", es_par_producto)

if __name__ == "__main__":
    main()
