# Definición de la función para agregar un número a la lista
def agregar_numero(lista_numeros, numero):
    lista_numeros.append(numero)  # Agrega el número a la lista
    print("Número agregado exitosamente.")  # Imprime un mensaje de confirmación

# Definición de la función para eliminar un número de la lista
def eliminar_numero(lista_numeros, numero):
    if numero in lista_numeros:  # Comprueba si el número está en la lista
        lista_numeros.remove(numero)  # Elimina el número de la lista
        print("Número eliminado exitosamente.")  # Imprime un mensaje de confirmación
    else:
        print("El número no está en la lista.")  
        # Si el número no está en la lista, imprime un mensaje indicando esto

# Definición de la función para mostrar la lista de números
def mostrar_lista(lista_numeros):
    print("Lista de números:", lista_numeros)  # Imprime la lista de números

# Definición de la función para ordenar la lista de números
def ordenar_lista(lista_numeros):
    lista_numeros.sort()  # Ordena la lista de números
    print("Lista de números ordenada:", lista_numeros)  # Imprime la lista de números ordenada

# Ejecución principal del programa
if __name__ == "__main__":
    lista_numeros = []  # Inicializa una lista vacía para almacenar números

    while True:  # Bucle infinito para mostrar el menú y realizar operaciones hasta que el usuario decida salir
        print("\n--- Menú ---")
        print("1. Agregar número")
        print("2. Eliminar número")
        print("3. Mostrar lista")
        print("4. Ordenar lista")
        print("5. Salir")

        opcion = input("Seleccione una opción: ")  # Solicita al usuario que seleccione una opción del menú

        if opcion == "1":
            numero = int(input("Ingrese el número a agregar: "))
            agregar_numero(lista_numeros, numero)  # Llama a la función para agregar un número
        elif opcion == "2":
            numero = int(input("Ingrese el número a eliminar: "))
            eliminar_numero(lista_numeros, numero)  # Llama a la función para eliminar un número
        elif opcion == "3":
            mostrar_lista(lista_numeros)  # Llama a la función para mostrar la lista de números
        elif opcion == "4":
            ordenar_lista(lista_numeros)  # Llama a la función para ordenar la lista de números
        elif opcion == "5":
            print("Saliendo del programa.")
            break  # Sale del bucle y termina la ejecución del programa si el usuario elige salir
        else:
            print("Opción inválida. Por favor, seleccione una opción válida.") 
         # Si el usuario ingresa una opción inválida, muestra un mensaje
