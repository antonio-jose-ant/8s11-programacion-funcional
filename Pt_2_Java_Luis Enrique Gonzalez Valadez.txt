package com.mycompany.yolanda2;

import java.util.Scanner;

/**
 *
 * @author jleon
 */
public class Yolanda2 {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        System.out.print("Ingresa un número: ");
        double num1 = scanner.nextDouble();
        System.out.print("Ingresa otro número: ");
        double num2 = scanner.nextDouble();

        realizarTodasLasOperaciones(num1, num2);
    }

    public static double suma(double a, double b) {
        return a + b;
    }

    public static double resta(double a, double b) {
        return a - b;
    }

    public static double multiplicacion(double a, double b) {
        return a * b;
    }

    public static void realizarTodasLasOperaciones(double num1, double num2) {
        double resultadoSuma = suma(num1, num2);
        double resultadoResta = resta(num1, num2);
        double resultadoMultiplicacion = multiplicacion(num1, num2);

        System.out.println("La suma es: " + resultadoSuma);
        System.out.println("La resta es: " + resultadoResta);
        System.out.println("La multiplicación es: " + resultadoMultiplicacion);    }
}
