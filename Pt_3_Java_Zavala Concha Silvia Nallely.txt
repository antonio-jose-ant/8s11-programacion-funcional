


import java.util.Scanner;

public class Prac1 {
    public static void main(String[] args) {
        
        Scanner scanner = new Scanner(System.in);

  
        System.out.println("Ingrese un dato:");
        String entrada = scanner.nextLine();


        String tipo;

        
        if (esEntero(entrada)) {
            tipo = "entero";
        } else if (esDecimal(entrada)) {
            tipo = "decimal";
        } else if (esBooleano(entrada)) {
            tipo = "booleano";
        } else {
            tipo = "cadena";
        }

       
        System.out.println("El dato ingresado es de tipo: " + tipo);

        
        scanner.close();
    }

    
    private static boolean esEntero(String str) {
        try {
         
            Integer.parseInt(str);
            return true; 
        } catch (NumberFormatException e) {
            return false; 
        }
    }

 
    private static boolean esDecimal(String str) {
        try {
          
            Double.parseDouble(str);
            return true; 
        } catch (NumberFormatException e) {
            return false; 
        }
    }

    
    private static boolean esBooleano(String str) {
       
        return str.equalsIgnoreCase("true") || str.equalsIgnoreCase("false");
    }
}
