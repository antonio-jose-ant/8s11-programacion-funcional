def es_numero_entero(cadena):
    try:
        int(cadena)
        return True
    except ValueError:
        return False

def es_numero_flotante(cadena):
    try:
        float(cadena)
        return True
    except ValueError:
        return False

def es_booleano(cadena):
    return cadena.lower() in ['true', 'false']

def main():
    entrada = input("Ingrese un dato: ")

    if es_numero_entero(entrada):
        tipo = "número entero"
    elif es_numero_flotante(entrada):
        tipo = "número flotante"
    elif es_booleano(entrada):
        tipo = "booleano"
    else:
        tipo = "cadena"

    print("El dato ingresado es de tipo:", tipo)

if __name__ == "__main__":
    main()
